4export function compose<C>(middlewares: Middleware<C>[]) {
5  return async function(ctx: C) {
6    function dispatch(i = 0): Promise<void> {
7      const fn = middlewares[i];
8      if (i >= middlewares.length) {
9        return Promise.resolve();
10      }
11      return Promise.resolve(fn(ctx, function next() {
12        return dispatch(i + 1);
13      }));
14    }
15    await dispatch();
16    return ctx;
17  };
18}
