44    const toJSON = () => {
45      return Array.from(formValues.entries()).reduce((prev, currentValue) => {
46        const [key, val] = currentValue;
47        Object.assign(prev, { [key]: val });
48        return prev;
49      }, {} as FormValues);
50    };

53        return formValues;
54      },

56      setItem: trailed((callback) => {
57        return (name, value, options) =>  {
58          const emit = options?.emit ?? false;
59          const remove = options?.remove ?? false;
60          formValues = new Map(formValues.entries());
61          if (remove) {
62            formValues.delete(name);
63          } else {
64            formValues.set(name, value);
65          }
66          setFormValues(formValues);
67          if (emit) callback(() => onChangeRef.current?.(toJSON()));
68        };
69      }),
70      reductions: new Map(),
71      invalidations: new Map(),
72      validators: new Map(),
73      status: new Map(),

78    if (fields.length) {
79      fields.forEach(key => context.reductions.get(key)?.());
80    } else {
81      context.reductions.forEach(v => v());
82    }
83  }, [context]);

85  const validate: FormInstance['validate'] = useCallback((...fields) => {
86    let values = context.toJSON();
87    let obs: Observable<Status>[];
88    if (fields.length) {
89      obs = fields.map(key => context.validators.get(key)).filter(Boolean).map(v => v!());
90      values = fields.reduce((previousValue, key) => Object.assign(previousValue, { [key]: values[key] }), {} as FormValues);
91    } else {
92      obs = Array.from(context.validators.values()).map(v => v());
93    }
94    return forkJoin(obs).pipe(
95      concatMap(results => {
96        if (results.every(v => v.state === 'valid')) {
97          return of(values);
98        }
99        return throwError(() => results.filter(v => v.state === 'invalid'));
100      }),
101    );
102  }, [context]);
103

104  const setItem: FormInstance['setItem'] = useCallback((...args) => {
105    return context.setItem(...args);
106  }, [context]);
107

108  const clearValidity: FormInstance['clearValidity'] = useCallback((...args) => {
109    if (args.length) {
110      args.forEach(name => context.invalidations.get(name)?.());
111    } else {
112      context.invalidations.forEach(v => v());
113    }
114  }, [context]);
115  
116  const instance: FormInstance = useMemo(() => {
117    return { setItem, reset, validate, clearValidity, values: formValues, toJSON: context.toJSON };
118  }, [formValues, reset, setItem, validate, context, clearValidity]);
119

120  const handleSubmit = useCallback((e: FormEvent) => {
121    e.preventDefault();
122    onSubmit?.(instance);
123  }, [instance, onSubmit]);
124

125  useImperativeHandle(ref, () => {
126    return instance;
127  });
128

129  return (
130    <form onSubmit={handleSubmit} className={combineClass(className)} style={style}>
131      <FormValuesContext.Provider value={formValues}>
132        <FormContext.Provider value={context}>{ children }</FormContext.Provider>
133      </FormValuesContext.Provider>
134    </form>
135  );
136}) as FormComponent;

34  const [status, store] = useStore<Status>(() => ({ state: 'initial', name }));
35

36  const nullVal = useMemo(() => {
37    if (children.type === 'input') {
38      return '';
39    }
40    return undefined;
41  }, [children]);
42  
43  const value = useMemo(() => {
44    return values?.get(name) ?? nullVal;
45  }, [name, nullVal, values]);
46

47  const status$ = useSubject<any>(action => action.pipe(
48    tap(() => store.set(Object.assign({}, store.state, { state: 'pending', name }))),
49    switchMap(value => {
50      if (value === restore) {
51        store.set({ state: 'initial', name });
52        return of(null).pipe(skip(1));
53      }
54      return validate(refs.current.rules, value);
55    }),
56    tap(() => store.set({ state: 'valid', name })),
57    store.capture(err => ({ state: 'invalid', name, message: err })),
58  ), [name, store]);
59

60  const onChange = useCallback((e: any) => {
61    const val = toFormValue(e);
62    status$?.next(val);
63    context?.setItem(name, val, { emit: true });
64    fieldChange?.(val);
65  }, [fieldChange, context, name, status$]);
66  
67  const clearValidity = useCallback(() => {
68    status$?.next(restore);
69  }, [status$]);
70

71  const reset = useCallback(() => {
72    const { context, defaultValue, name } = refs.current;
73    const value = context?.values.get(name) ?? nullVal;
74    const next = toFormValue(defaultValue) ?? nullVal;
75    clearValidity();
76    if (value !== next) {
77      context?.setItem(name, next, { emit: true });
78    }
79  }, [clearValidity, nullVal]);
80  
81  const validateField = useCallback(() => {
82    status$?.next(value); // The status will be updated to pending synchronously
83    return store.state$.pipe(
84      skipWhile(v => v.state === 'pending'),
85      take(1),
86    );
87  }, [status$, value, store]);
88  
89  useEffect(() => {
90    context?.validators.set(name, validateField);
91    return () => {
92      context?.validators.delete(name);
93    };
94  }, [context, name, validateField]);
95

96  useEffect(() => {
97    context?.invalidations.set(name, clearValidity);
98    return () => {
99      context?.invalidations.delete(name);
100    };
101  }, [context, name, clearValidity]);
102  
103  useEffect(() => {
104    context?.reductions.set(name, () => {
105      reset();
106    });
107    return () => {
108      context?.reductions.delete(name);
109    };
110  }, [context, name, reset]);
111

112  useEffect(() => {
113    context?.status.set(name, status);
114    return () => {
115      context?.status.delete(name);
116    };
117  }, [context, status, name]);
118

119  useEffect(() => {
120    const { context } = refs.current;
121    if (refs.current.name !== name) {
122      context?.setItem(refs.current.name, null, { emit: true, remove: true });
123    }
124    refs.current.name = name;
125    reset();
126  }, [reset, name]);
127  
128  useEffect(() => {
129    onStatusChange?.(status);
130  }, [onStatusChange, status]);
131

132  return cloneElement(children, {

134    value,
135  });

140function validate(rules: Rule[], value: any) {
141  return concat(
142    ...rules.map(v => {
143      let obs: Observable<null | string>;
144      if (typeof v === 'function') {
145        const ret = v(value);
146        if (isObservable(ret)) {
147          obs = ret;
148        } else if (ret instanceof Promise) {
149          obs = from(ret);
150        } else {
151          obs = of(ret);
152        }
153      } else {
154        obs = v.required && !value ? of(v.message ?? '') : of(null);
155      }
156      obs = obs.pipe(
157        concatMap(res => res === null ? of(null) : throwError(() => res))
158      );
159      return obs as Observable<null>;
160    }),
161    of(null),
162  ).pipe(takeLast(1));
163}

165function toFormValue(e: any) {
166  if (e && e.target && Object.hasOwn(e.target, 'value')) {
167    return e.target.value;
168  }
169  return e;
