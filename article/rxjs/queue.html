固定任务数量

1import { interval, merge, take, tap } from 'rxjs';
2

3const count1 = interval(1000).pipe(
4  take(10),
5  tap(v => console.log(`count1:${v}`)),
6);
7

8const count2 = interval(3000).pipe(
9  take(2),
10  tap(v => console.log(`count2:${v}`)),
11);
12

13const count3 = interval(3000).pipe(
14  take(5),
15  tap(v => console.log(`count3:${v}`)),
16);
17

18const count4 = interval(2000).pipe(
19  take(10),
20  tap(v => console.log(`count4:${v}`)),
21);
22

23const count5 = interval(5000).pipe(
24  take(5),
25  tap(v => console.log(`count5:${v}`)),
26);
27

28export function startQueue() {
29  // 表示同时执行以上所有observable，但最大队列数为2
30  return merge(count1, count2, count3, count4, count5, 2).subscribe();
31}

点击开始执行startQueue，打开f12查看打印结果

动态任务数量

1import { fromEvent, mergeScan, tap, timer, map } from 'rxjs';
2

3export function dynamicQueue(dom: HTMLElement) {
4  return fromEvent(dom, 'click').pipe(
5    mergeScan((acc, value, index) => {
6      const duration = Math.random() * 10;
7      return timer(duration * 1000).pipe(
8        map(() => acc + 1),
9        tap(() => console.log(`第${index + 1}个下载完毕，用时${duration}秒`)),
10      );
11    }, 0, 2), // 0为seed，2为最大队列数
12  ).subscribe();
13}

连续点击下载创建多个下载任务，模拟最大并发下载为2的场景
