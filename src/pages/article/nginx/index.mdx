# 配置参考

```
user  root; # 运行用户
worker_processes  auto; # Nginx进程，一般设置为和CPU核数一样


events {
    worker_connections  1024; # 单个后台进程的最大并发数
}

http {
    include       /usr/local/nginx/conf/mime.types; # 文件扩展名与类型映射表
    default_type  application/octet-stream; # 默认文件类型

    # 设置日志模式
    log_format  main  '$remote_addr - $remote_user [$time_local] "$request" '
                      '$status $body_bytes_sent "$http_referer" '
                      '"$http_user_agent" "$http_x_forwarded_for"';

    sendfile        on;
    server_tokens   off;

    keepalive_timeout  65;

    gzip              on;
    gzip_types        application/javascript text/css;
    gzip_min_length   2048;

    map $http_upgrade $connection_upgrade {
      default upgrade;
      '' close;
    }

    add_header Cache-Control no-cache; # 协商缓存

    server {
      listen 80 default_server;
      absolute_redirect off;
      index index.html;
      root /jumpserver-dashboard/; # 静态文件所在目录

      location / {
        try_files $uri /index.html; # 静态资源配置（单页面程序）
      }

      location /api {
        proxy_pass       http://localhost:5000/; # 将 /api/* 请求转发至 http://localhost:5000/*
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header Host $host;
      }
    }
}
```
